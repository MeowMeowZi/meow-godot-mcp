        // Extract args, build text, return messages array

        });
    }
}

      Tool: set_keyframe

      Tool: set_animation_properties
      Parameters: { "animation_name": "fade_in", "duration": 0.5, "loop_mode": 0, "player_path": "/root/Scene/UI/AnimationPlayer" }
      Result: Animation is 0.5s long, plays once (no loop)

    Step 6: Verify the animation
      Tool: get_animation_info
      Parameters: { "player_path": "/root/Scene/UI/AnimationPlayer" }
      Result: Confirm animation list, tracks, keyframes, and properties
